Table
=======
CREATE TABLE "SYSTEM"."REALTIME_CUSTOMER" 
   (	"CUSTID" NUMBER NOT NULL ENABLE, 
	"CUSTNAME" VARCHAR2(20 BYTE), 
	"CUSTADD" VARCHAR2(20 BYTE), 
	"PRINCIPLE" FLOAT(126), 
	"TIME" FLOAT(126), 
	"RATE" FLOAT(126), 
	"INTRESTAMOUNT" FLOAT(126), 
	 CONSTRAINT "REALTIME_CUSTOMER_PK" PRIMARY KEY ("CUSTID")
	 
Sequence
========
CREATE SEQUENCE  "SYSTEM"."CUSTID_SEQ"  MINVALUE 1000 MAXVALUE 10000 INCREMENT BY 1 START WITH 1000
